 How to use on another machine

  # 1. Copy .env.example to .env
  cp .env.example .env

  # 2. Edit .env - fill in your values (at minimum change MYSQL_ROOT_PASSWORD and SECRET_KEY)
  nano .env

  # 3. Start everything
  docker compose up -d --build
  backend: uv run uvicorn app.main:app --reload 
  frontend: npm run dev
